#include <iostream>
using namespace std;
template <typename T>
T func(T num1) { num1++; return num1; }
template <typename T>
T func(T num1, T num2)
{
	T Min;
	if (num1 > num2) { Min = num2; }
	else { return Min = num1; }
	return Min;
}
template <typename T>
T func(T num1, T num2, T num3)
{
	T Max;
	if (num1 > num2 && num1 > num3)
		Max = num1;
	else if (num2 > num1 && num2 > num3)
		Max = num2;
	else if (num3 > num1 && num3 > num2)
		Max = num3;
	return Max;
}
void main()
{
	int num1, num2, num3;
	cout << "정수 하나 입력 : ";
	cin >> num1;
	cout << func(num1) << endl;
	cout << "정수 두개 입력 : ";
	cin >> num1 >> num2;
	cout << "최소 값은 : " << func(num1, num2) << "입니다." << endl;
	cout << "정수 세개 입력 : ";
	cin >> num1 >> num2 >> num3;
	cout << "최대 값은 "<<func(num1, num2, num3) << "입니다 .";
}
