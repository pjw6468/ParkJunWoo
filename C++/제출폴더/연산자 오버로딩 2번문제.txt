#include <iostream>
#include <Windows.h>
using namespace std;
class Time
{
private:
	int m_iHour;
	int m_iMin;
public:
	void ShowTime();
	void StudyTimeSet(int Hour,int Min);
	Time operator +(Time time);
	Time() {};
	Time(int Hour, int Min) { m_iHour = Hour; m_iMin = Min; }
	~Time() {};
};
void Time::ShowTime()
{
	cout << "총 공부 시간 : " << m_iHour << " : " << m_iMin << endl;
}
void Time::StudyTimeSet(int Hour, int Min)
{
	m_iHour = Hour;
	m_iMin = Min;
}
Time Time::operator +(Time time)
{
	time.m_iHour = this->m_iHour + time.m_iHour;
	time.m_iMin = this->m_iMin + time.m_iMin;
	if (time.m_iMin >= 60){while (time.m_iMin >= 60){time.m_iMin -= 60;time.m_iHour += 1;}}
	return time;
}
void main()
{
	int Day = 0;
	int Hour;
	int Min;
	int Select;
	Time Study(0,0), Plus(0,0);
	while (1)
	{
		system("cls");
		Study.ShowTime();
		cout << "==== 공부 시간 관리 프로그램<" << Day << "Day>====" << endl;
		cout << "1.시간 등록 " << endl << "2. 종료 " << endl << "입력 : ";
		cin >> Select;
		switch (Select)
		{
		case 1:
			cout << "시간 : ";
			cin >> Hour;
			cout << "분 : ";
			cin >> Min;
			Plus.StudyTimeSet(Hour, Min);
			Study = Study + Plus;
			Day++;
			break;
		case 2:
			return;
		}
	}
}
