A operator / (A a, A b)
{
	if (a.x > b.x) { a.x = a.x/b.x; }
	else { a.x= b.x/a.x; }
	if (a.y > b.y) { a.y = a.y/b.y; }
	else { a.y = b.y / a.y; }
	return a;
}
